pwd لعرض مسار الملف

cd /file name/new file       علي سبيل المثال file name داخل المجلد new file للذهاب الى المجلد 

cd ~                         الذهاب لمجلد المنزل

cd ..                        للعودة الى المجلد الاب

ls                           عرض الملفات والمجلدات التي في داخل المسار الحالى

ls -l                        لعرض معلومات تفصيلية اكثر عن المجلدات والملفات التى في داخل المسار الحالى

ls -a                        لعرض الملفات والمجلدات داخل المسار الحالي بالاضافة الى الملفات والمجلدات المخفية

ls -la                       فى الاعلى ls -a و ls -l اندماج  

ls /file_name                لعرض الملفات التي في مجلد مختلف عن المسار الحالى

file /file_name              /file_name لعرض تفاصيل موجزة عن محتويات ملف معين علي سبيل المثال

less /text file directory    هو مسار الملف النصي text file direcory لعرض محتويات ملف نصي حيث 

ls a*                        a يقوم بعرض جميع الملفات التي تبدا بحرف 

ls f*.txt                    .txt ر وتنتهي بالاحقة f يقوم بعرض جميع الملفات التي تبدا بحرف 

ls *.html                    .html عرض جميع الملفات التي تنتهي بالاحقة 

ls image?                    يليها حرف واحد image يعرض اي ملف يبدا بالكلمة 

ls imag???                   ويليها ثلالة محارف image يعرض اي ملف يبدا بالكلمة 

ls [characters]*             [] يعرض اي ملف يبدا باحد الحروف التي في داخل القوسين 

ls [!abc]*                   عرض اي ملف لا يبدا بالاحرف التي بين القوسين بعد علامة التعجب

ls etc/pass*                 etc داخل الملف pass عرض الملفات التي تبدا بالكلمة 

ls /etc/*.conf               etc داخل الملف .conf عرض الملفات التي تبدا بالاحقة 

mkdir file_name              داخل المسار الحالي file_name ينشئ ملف باسم 

mkdir -p a/b/c               c بداخله المجلد b بداخله المجلد a انشاء المجلد

mkdir .test                  لان اسم الملف يبدا بنقطة test انشاء مجلد مخفي اسمه 

cp <file_name> <directory>   نسخ الملف الي المسار المحدد

cp -a                       نسخ الملفات والمجلدات مع كل خاصياتهم بما فيها الاوذونات والمجموعة المالكة

cp -i                        طلب موافقة المستخدم قبل استبدال ملف موجود مسبقا.
 
cp -v                        لإظهار معلومات حول تقدم عملية النسخ

cp -r folder_name folder_dir نسخ المجلدات بما فيها الي المسار المحدد        

mv <file_name> <folder_name>  نقل ملف الى مجلد اذا تم تحديد اسم مجلد غير موجود لنقل الملف اليه سيتغير اسم الملف حسب الاسم الذي تم تحديده

rm  <file_name>               حذف الملف

rm -i <file_name>             لاظهار تنبيه قبل الحذف

rm -r <file_name>             لحذف المجلدات

rm -v <file_name>             لاظهار معلومات عن تقدم عملية الحذف

type <command>               الامر الذي نريد معرفة نوعه <command> نضيف مكان 

help <command>               عرض الاوامر المضمننة في الصدفة

<command> --help             عرض معلومات تفصيلية عن امر معين

clear                        تنظيف محاكي الطرفية

man <command>                مشاهدة صفحة الدليل لأمر معين

info <command>               لامر معين info مشاهدة صفحة المعلومات 


                   Nano commands

nano <file_name>.txt                                 إنشاء ملف نصي جديد

nano +<row_number>,<column_number> <file_name>       نضيف مكان <row_number> رقم السطر، ومكان <column_number> رقم العمود ومكان <file_name> اسم الملف.الانتقال إلى سطر وعمود معين

البحث عن كلمة أو جملة معينة نضغط على `Ctrl + w` في لوحة المفاتيح.
لحفظ التغييرات نضغط على `Ctrl + o` في لوحة المفاتيح.
للخروج من الملف نضغط على `Ctrl + x` في لوحة المفاتيح.
محاذاة الأسطر في شاشة الطرفية نضغط على `Ctrl + j` في لوحة المفاتيح.
إدراج محتوى ملف نصي آخر ضمن الملف الحالي نضغط على `Ctrl + r` في لوحة المفاتيح.
قص سطر معين نضغط على `Ctrl + k` في لوحة المفاتيح.
لصق سطر معين نضغط على `Ctrl + u` في لوحة المفاتيح.
الانتقال إلى سطر وعمود محدد نضغط على `_ + Ctrl` في لوحة المفاتيح.
استبدال كلمة بكلمة أخرى نضغط على `/ + Ctrl` في لوحة المفاتيح.


                                 اعادة التوجيه 



<command> > <file_name>         حفظ مخرجات أمر إلى ملف معين

less <file_name>                لإظهار محتويات ملف نصي

<command> <file_directory> > <file_name>      حفظ مخرجات أمر إلى ملف معين في مسار مختلف عن المسار الحالي

<command> >> <file_name>                     إضافة مخرجات أمر إلى ملف معين بدون حذف محتويات الملف

<command> /bin/usr 2> <file_name>            إعادة توجيه المخرجات الناتجة من الخطأ عند تطبيق أمر وكتابتها إلى ملف معين

<command> <file_directory> &> <file_name>    إعادة توجيه مجرى الخرج و الخطأ معاً إلى ملف واحد:
